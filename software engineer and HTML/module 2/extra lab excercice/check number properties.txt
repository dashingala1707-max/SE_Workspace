#include <stdio.h>

int main() {
    int num;

    // Prompt user for input
    printf("Enter an integer: ");
    // Use the scanf function to read the integer from the user
    scanf("%d", &num);

    // --- Check if the number is even or odd using the modulo operator (%) ---
    if (num % 2 == 0) {
        printf("%d is even.\n", num);
    } else {
        printf("%d is odd.\n", num);
    }

    // --- Check if the number is positive, negative, or zero using comparison operators (<, >, ==) ---
    if (num > 0) {
        printf("%d is positive.\n", num);
    } else if (num < 0) {
        printf("%d is negative.\n", num);
    } else {
        printf("%d is zero.\n", num);
    }

    // --- Check if the number is a multiple of both 3 and 5 using modulo (%) and logical AND (&&) operators ---
    if (num % 3 == 0 && num % 5 == 0) {
        printf("%d is a multiple of both 3 and 5.\n", num);
    } else {
        printf("%d is not a multiple of both 3 and 5.\n", num);
    }

    return 0;
}
