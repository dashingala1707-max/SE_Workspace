#include <stdio.h>

int main() {
    float marks;

    // Taking marks as input from the user
    printf("Enter the marks of the student: ");
    scanf("%f", &marks);

    // Decision-making process using if-else
    if (marks > 90) {
        printf("Grade: A\n");
    } 
    else if (marks > 75 && marks <= 90) {
        printf("Grade: B\n");
    } 
    else if (marks > 50 && marks <= 75) {
        printf("Grade: C\n");
    } 
    else if (marks <= 50) {
        printf("Grade: D\n");
    }
    else {
        // This case handles unexpected inputs (e.g., non-numeric)
        printf("Invalid Input\n");
    }

    return 0;
}
