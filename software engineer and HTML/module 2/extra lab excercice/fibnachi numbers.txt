#include <stdio.h>
#include <time.h>

// Recursive function to calculate the Nth Fibonacci number
long long int fibonacci_recursive(int n) {
    if (n <= 1) {
        return n; // Base cases: F(0) = 0, F(1) = 1
    }
    return fibonacci_recursive(n - 1) + fibonacci_recursive(n - 2); // Recursive step
}

// Iterative function to calculate the Nth Fibonacci number
long long int fibonacci_iterative(int n) {
    if (n <= 1) {
        return n;
    }
    long long int a = 0, b = 1, next_term;
    for (int i = 2; i <= n; ++i) {
        next_term = a + b;
        a = b;
        b = next_term;
    }
    return b; // The nth term
}

int main() {
    int N;
    printf("Enter the value of N to find the Nth Fibonacci number: ");
    scanf("%d", &N);

    if (N < 0) {
        printf("Fibonacci of a negative number is not possible.\n");
        return 1;
    }

    clock_t start, end;
    double cpu_time_used;
    long long int result_recursive, result_iterative;

    // Measure time for the recursive method
    start = clock();
    result_recursive = fibonacci_recursive(N);
    end = clock();
    cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
    printf("\nRecursive Method:\n");
    printf("Fib(%d) = %lld\n", N, result_recursive);
    printf("Time taken: %f seconds\n", cpu_time_used);

    // Measure time for the iterative method
    start = clock();
    result_iterative = fibonacci_iterative(N);
    end = clock();
    cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
    printf("\nIterative Method:\n");
    printf("Fib(%d) = %lld\n", N, result_iterative);
    printf("Time taken: %f seconds\n", cpu_time_used);

    return 0;
}
