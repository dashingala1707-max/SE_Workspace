#include <stdio.h>

int main() {
    int i = 1; // Initialize the loop counter

    printf("Numbers from 1 to 10 using a do-while loop:\\n");
    // Execute the body, then check the condition
    do {
        printf("%d ", i);
        i++; // Increment the counter
    } while (i <= 10); // Continue looping as long as i is less than or equal to 10
    printf("\\n");

    return 0;
}
