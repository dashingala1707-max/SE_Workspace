#include<iostream>
using namespace std;

int main()
{
    int a = 10;
    int b = 20;
    char c = 't';
    char d = 'f';
    
    cout<<"-------arithamatic operators-----------"<<endl;
    cout<<" a + b = "<<a + b<<endl;
    cout<<" a - b = "<<a - b<<endl;
    cout<<" a * b = "<<a * b<<endl;
    cout<<" a / b = "<<a / b<<endl;
    cout<<" a % b = "<<a % b<<endl;
    
    cout<<"------------relational operators----------------"<<endl;
    cout<<"a > b = "<<(a > b)<<endl;
    cout<<"a < b = "<<(a < b)<<endl;
    cout<<"a >= b = "<<(a >= b)<<endl;
    cout<<"a <= b = "<<(a <= b)<<endl;
    cout<<"a == b = "<<(a == b)<<endl;
    cout<<"a != b = "<<(a != b)<<endl;
    
    cout<<"---------------logical operators---------------"<<endl;
    cout<<"c && d is"<<(c && d)<<endl;
    cout<<"c || d is"<<(c || d)<<endl;
    cout<<"! c is"<<(c)<<endl;
    
}
