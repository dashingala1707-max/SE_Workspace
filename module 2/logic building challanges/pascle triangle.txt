#include <stdio.h>

// Recursive function to calculate the factorial of a number
unsigned long long factorial(int n) {
    if (n <= 1) {
        return 1; // Base case for recursion
    }
    return n * factorial(n - 1);
}

// Function to calculate binomial coefficient C(n, r)
unsigned long long combinations(int n, int r) {
    if (r < 0 || r > n) {
        return 0; // Invalid combination
    }
    // C(n, r) = n! / (r! * (n-r)!)
    return factorial(n) / (factorial(r) * factorial(n - r));
}

int main() {
    int N, i, j;

    printf("Enter the number of rows: ");
    scanf("%d", &N);

    for (i = 0; i < N; i++) {
        // Print leading spaces for alignment
        for (int space = 0; space < (N - i - 1) * 2; space++) {
            printf(" ");
        }

        for (j = 0; j <= i; j++) {
            // Calculate and print the value for each position using the recursive function
            printf("%4llu", combinations(i, j));
        }
        printf("\n"); // Move to the next row
    }

    return 0;
}
