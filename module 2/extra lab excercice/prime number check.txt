#include <stdio.h>
#include <math.h> // Required for the sqrt() function

// Function to check if a number is prime
int isPrime(int n) {
    if (n <= 1) return 0; // Not prime
    // Check for factors up to the square root of n
    for (int i = 2; i <= sqrt(n); i++) {
        if (n % i == 0) {
            return 0; // Not prime
        }
    }
    return 1; // Prime
}

int main() {
    int limit, i;

    printf("Enter the upper limit: ");
    scanf("%d", &limit);

    printf("Prime numbers between 1 and %d are: ", limit);

    // Iterate from 2 up to the limit
    for (i = 2; i <= limit; i++) {
        if (isPrime(i)) {
            printf("%d ", i);
        }
    }

    printf("\\n");
    return 0;
}
