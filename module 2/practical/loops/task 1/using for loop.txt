#include <stdio.h>

int main() {
    int i;

    printf("Numbers from 1 to 10 using a for loop:\\n");
    // Initialize i to 1, continue as long as i is less than or equal to 10, increment i after each iteration
    for (i = 1; i <= 10; ++i) {
        printf("%d ", i);
    }
    printf("\\n");

    return 0;
}
