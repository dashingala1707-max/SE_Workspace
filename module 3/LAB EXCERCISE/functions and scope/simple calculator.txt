int main()
{
    int num1,num2;
    char ab;
    int c = 0;
    
    cout<<"enter two numbers : ";
    cin>>num1>>num2;
    
    cout<<"enter the operator(+,-,*,/) : ";
    cin>>ab;
    
    if(ab == '+')
    {
        c = num1 + num2;
    }
    else if(ab == '-')
    {
        c = num1 - num2;
    }
    else if(ab == '*')
    {
        c = num1 * num2;
    }
    else if(ab == '/')
    {
        c = num1 / num2;
    }
    else
    {
        cout<<"you entered wrong operator";
    }
    cout<<"your answer is : "<< c << endl;
    
    
    return 0;
}
