	#include<iostream>
using namespace std;

int main()
{
    int marks;
    
    cout<<"enter the marks :";
    cin>>marks;
    
    if(marks >= 90)
    {
        cout<<"A grade"<<endl;
    }
    else if(marks >= 70)
    {
        cout<<"B grade"<<endl;
    }
    else if(marks >= 60)
    {
        cout<<"C grade"<<endl;
    }
    else if(marks >= 50)
    {
        cout<<"D grade"<<endl;
    }
    else
    {
        cout<<"FAIL"<<endl;
    }
    
    return 0;
}
